#version 450 core

layout(triangles) in;
layout(triangle_strip, max_vertices = 3) out;

in vec4 vertColor[];

out vec4 fragColor;

void main() {
    bool skip = false;
    for (int i = 0; i < 3; i++) {
        if (gl_in[i].gl_Position.z <= 0)
            skip = true;
    }
    if (skip)
        return;

    for (int i = 0; i < 3; i++) {
        gl_Position = gl_in[i].gl_Position;
        fragColor = vertColor[i];
        EmitVertex();
    }
    EndPrimitive();
}
